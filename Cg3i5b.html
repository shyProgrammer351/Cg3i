#include <stdio.h> 
                 
 printMap( int map[7][9] ) { 
	char symbol[10][1024] = { "  ", " \033[44m   \033[0m", " \033[34m문 \033[0m" }; 
	printf("\033[2K\033[2;1f"); 
	 ( int row = 0; row < 7; row++ ) { 
		 ( int column = 0; column < 9; column++ ) 
			printf( "%s", symbol[map[row][column]] ); 
		printf( "\n" ); 
	} 
} 
 
 move( char direction, row, column, int map[7][9] ) { 
	 ( ( ( direction == 'w' ) || ( direction == 'W' ) )  ( row > 0 )  ( map[ row-1 ][ column ] != 1 ) ) 
		row = row-1; 
	 ( ( ( direction == 'a' ) || ( direction == 'A' ) )  ( column > 0 )  ( map[ row ][ column-1 ] != 1 ) ) 
		column = column-1; 
	 ( ( ( direction == 's' ) || ( direction == 'S' ) )  ( row < 6 )  ( map[ row+1 ][ column ] != 1 ) ) 
		row = row+1; 
	 ( ( ( direction == 'd' ) || ( direction == 'D' ) )  ( column < 8 )  ( map[ row ][ column+1 ] != 1 ) ) 
		column = column+1; 
} 
 
int finish( int here ) { 
	( here == 2 ) { 
		printf( " \033[2J \033[4;15fSUCCESS!\033[7;15f" ); 
		 1; 
	} 
	 { 
		 0; 
	} 
}
