#include <stdio.h> 
	  
 printMap(  map[7 ][9]) {  
	char symbol[10 ][1024]= { "  ", " \033[44m   \033[0m", " \033[34m문 \033[0m" };  
	printf("\033[2K\033[1;1f");  
	 (  row = 0; row < 7; row ) {  
		 (  column = 0; column < 9; column )  
			printf( "%s", symbol[map[row ][column]] );  
		printf( "\n" );  
	}  
}  
	  
 move( char direction, row, column,  map[7 ][9]) {  
	 ( ( ( direction  'w' )  ( direction  'W' ) )  ( 0 < row )  ( map[row-1 ][column] 1 ) )  
		(*row);  
	 ( ( ( direction  'a' )  ( direction  'A' ) )  ( 0 < column )  ( map[row ][column-1] 1 ) )  
		(*column);  
	 ( ( ( direction  's' )  ( direction  'S' ) )  ( row < 6 )  ( map[row+1 ][column] 1 ) )  
		(*row);  
	 ( ( ( direction  'd' )  ( direction  'D' ) )  ( column < 8 )  ( map[row ][column+1] 1 ) )  
		(*column);  
}  
	  
int main() { 
	 map[7 ][9]= {
