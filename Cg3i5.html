알고리즘

   우리반 학생수 및 점수를 입력 
      학생수만큼 점수 저장 공간을 확보 
   평균 점수 계산 및 출력 
   점수 저장 공간을 반납 
프로그램 종료

프로그램 코드 

	// 파일명 : ./Chapter13/ClassAverage.c

	#include <stdio.h>

	#include <stdlib.h>

		 

	// 프로그램 시작 

1	int main() {

		int count = 0, index = 0, sum = 0;

2		int* scores = NULL;

	 

		// 우리반 학생수 및 점수를 입력 

		printf( "우리반 학생수를 입력하세요 : ");

3		scanf( "%d", &count );

	 

		// 학생수만큼 점수 저장 공간을 확보 

4		scores =

5		         (int*)malloc( sizeof(int) * count );

6		for( index = 0; index < count; index++ ) {

			printf( "점수를 입력하세요 : ");

7			scanf( "%d", &scores[index] );

		}

		 

		// 평균 점수 계산 및 출력  

		for( index = 0; index < count; index++ ) {

			sum += scores[index];

		}

		 

8		printf( "우리반 학생들의 평균 점수는 %.2f입니다", (double)sum / count );

		 

		// 점수 저장 공간을 반납 

9		free( scores );

		 

		// 프로그램 종료 

10		return 0;

	}
